**문제** : https://www.acmicpc.net/problem/17835

**풀이** : 면접장을 출발점으로 잡아 연결된 단뱡향 도로를 반대로 방향을 바꿔서 면접장에서 갈수있는 도시의 각각의 거리를 다익스트라를 통해서 구해주면된다.

**소스코드** :

```c++
#include<bits/stdc++.h>

using namespace std;

typedef long long ll;
const int MAXN = 1e5 + 1;
const ll INF = 1e17;
typedef pair<ll, ll> pi;
int n, m, k;
vector<pi> adj[MAXN];
bool visit[MAXN];
ll dist[MAXN];

struct cmp {
	bool operator()(pi a, pi b) {
		return a.second > b.second;
	}
};

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);

	cin >> n >> m >> k;

	while (m--) {
		ll u, v, c;
		cin >> u >> v >> c;
		adj[v].push_back({ u,c });
	}

	for (int i = 0; i <= n; i++) {
		dist[i] = INF;
	}

	priority_queue<pi, vector<pi>, cmp> pq;

	for (int i = 0; i < k; i++) {
		int a;
		cin >> a;
		pq.push({ a,0 });
		dist[a] = 0;
	}

	while (!pq.empty()) {
		ll cur = pq.top().first;
		ll cost = pq.top().second;
		pq.pop();
		if (visit[cur]) continue;

		visit[cur] = true;

		for (auto next : adj[cur]) {
			if (dist[next.first] > cost + next.second) {
				dist[next.first] = cost + next.second;
				pq.push({ next.first,dist[next.first] });
			}
		}
	}

	ll ans = -INF;
	ll idx = 0;

	for (int i = 1; i <= n; i++) {
		if (ans < dist[i]) {
			ans = dist[i];
			idx = i;
		}
	}

	cout << idx << "\n" << ans << "\n";

	return 0;
}
```
