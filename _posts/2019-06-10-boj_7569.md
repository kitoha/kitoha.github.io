문제:  https://www.acmicpc.net/problem/7569

높이 h m은 상자의 가로 칸 n은 상자의 세로 칸인 창고의 토마토가 모두 익을 때 까지 필요한 날짜를 구하는 문제이다.

풀이:

3차원 배열 [h][m][n]을 구성하여 넓이 우선 탐색(bfs)를 이용하여 토마토가 있는 구간에서 최대 값을 구하면 해결할 수 있다.

모든 토마토가 익지 못하였을 경우는 -1를 출력해주면 된다.

~~~c++
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<queue>
#define R 101

using namespace std;

struct point {
   int u, v,z, cnt;
   point(int _u, int _v,int _z, int _cnt) :u(_u), v(_v),z(_z), cnt(_cnt) {};
};

int dx[7] = { -1,1,0,0,0,0,0 };
int dy[7] = { 0,0,-1,1,0,0,0 };
int dz[7] = { 0,0,0,0,-1,1,0 };
int n, m, h,val;
int a[R][R][R],visit[R][R][R];
queue<point> q;

int main() {

   memset(visit, -1, sizeof(visit));

   scanf("%d%d%d", &n, &m, &h);

   for (int k = 0; k < h; k++) {
      for (int i = 0; i < m; i++) {
         for (int j = 0; j < n; j++) {
            scanf("%d", &a[k][i][j]);
            if (a[k][i][j] == 1) {
               q.push({ i,j,k,0 });
               visit[k][i][j] = 0;
            }

            if (a[k][i][j] == 0) val++;
         }
      }
   }

   int ans = 0;
   int c = 0;

   while (!q.empty()) {
      int x = q.front().u;
      int y = q.front().v;
      int z = q.front().z;
      int cnt = q.front().cnt;
      q.pop();

      for (int i = 0; i <= 5; i++) {
         int nx, ny, nz;
         nx = x + dx[i];
         ny = y + dy[i];
         nz = z + dz[i];

         if (nx < 0 || nx >= m || ny < 0 || ny >= n || nz < 0 || nz >= h) continue;
         if (visit[nz][nx][ny] != -1 || a[nz][nx][ny] != 0) continue;

         visit[nz][nx][ny] = cnt + 1;
         c++;
         ans = max(ans, visit[nz][nx][ny]);
         q.push({ nx,ny,nz,cnt + 1 });
      }
   }

   if (c != val) printf("-1\n");
   else printf("%d\n", ans);

   return 0;
}
~~~
