**문제** : https://www.acmicpc.net/problem/17836

**풀이** :
그람을 먹었을 때와 그람을 먹지 않았을때 두 경우로 나누어서 BFS 탐색을 하면된다.

**소스코드** :

```c++
#include<bits/stdc++.h>

using namespace std;

const int MAXN = 101;
struct point {
	int y, x, status, cnt;
	point(int _y, int _x, int _status,int _cnt) :y(_y), x(_x), status(_status),cnt(_cnt) {};
};
int n, m, t;
int cells[MAXN][MAXN];
bool visit[2][MAXN][MAXN];
int dy[] = { -1,1,0,0 };
int dx[] = { 0,0,-1,1 };

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);

	cin >> n >> m >> t;

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			cin >> cells[i][j];
		}
	}

	queue<point> q;
	q.push(point(0, 0, 0, 0));

	while (!q.empty()) {
		int y = q.front().y;
		int x = q.front().x;
		int status = q.front().status;
		int cnt = q.front().cnt;
		q.pop();

		if (cnt > t) {
			break;
		}

		if (y == n - 1 && x == m - 1) {
			cout << cnt << "\n";
			return 0;
		}

		for (int i = 0; i <= 3; i++) {
			int ny = y + dy[i];
			int nx = x + dx[i];
			int tmp = status;
			if (ny < 0 || ny >= n || nx < 0 || nx >= m) continue;

			if (visit[status][ny][nx]) continue;
			if (status != 1 && cells[ny][nx] == 1) continue;

			visit[status][ny][nx] = true;

			if (tmp==0 && cells[ny][nx] == 2) tmp = 1;

			q.push(point(ny, nx, tmp, cnt + 1));

		}
	}

	cout << "Fail\n";

	return 0;
}
```
